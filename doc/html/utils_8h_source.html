22 /*
23  * Macros and functions useful to check whether sockets are valid. We
24  * use the same definition of sockets as libevent, i.e., a `int` on Unix
25  * and a `intptr_t` on Win32.
26  */
27 
28 /*
29  * To mark sockets as invalid. Strictly speaking on Windows it should
30  * be INVALID_SOCKET, which however is an unsigned constant. Since
31  * libevent uses inptr_t for sockets, here we use -1 to indicate that
32  * a socket is not valid, to avoid compiler warnings.
33  *
34  * See also the much longer comment on this topic below.
35  */
36 #define IGHT_SOCKET_INVALID -1
37 
38 static inline int
39 ight_socket_valid_win32_(intptr_t filenum)
40 {
41  /*
42  * On Windows there is only one value marking the socket as invalid,
43  * all the other values are to be considered valid sockets:
44  *
45  * "Windows Sockets handles have no restrictions, other than that
46  * the value INVALID_SOCKET is not a valid socket. Socket handles may
47  * take any value in the range 0 to INVALID_SOCKET–1".
48  *
49  * http://goo.gl/FTesjR (msdn.microsoft.com)
50  *
51  * This explains why libevent on Windows defines the socket to be
52  * intptr_t rather than uintptr_t: because they use a negative
53  * value, -1, to represent the invalid socket.
54  *
55  * For reference, wine's winsock.h defines in fact SOCKET as
56  * UINT_PTR and INVALID_SOCKET as (~0):
57  *
58  * http://goo.gl/Mo4ReU (github.com/wine-mirror/wine)
59  *
60  * Note that the unsigned value ~0 is equal to -1 if converted to a
61  * signed value of the same size (i.e., uintptr_t => intptr_t).
62  *
63  * To conclude, only -1 (equal to ~0) is invalid socket, hence
64  * the check below:
65  */
66  return (filenum != IGHT_SOCKET_INVALID);
67 }

69 static inline int
70 ight_socket_valid_unix_(int filenum)
71 {
72  return (filenum >= 0 && filenum <= INT_MAX);
73 }

75 static inline int
76 ight_socket_valid(evutil_socket_t filenum)
77 {
78 #ifdef WIN32
79  return (ight_socket_valid_win32_(filenum));
80 #else
81  return (ight_socket_valid_unix_(filenum));

85 static inline intptr_t
86 ight_socket_normalize_if_invalid_win32_(intptr_t filenum)
87 {
88  /*
89  * Nothing do to; on Windows there is a single, canonical
90  * representation of the invalid socket: ~0.
91  */
92  return (filenum);
93 }

95 static inline int
96 ight_socket_normalize_if_invalid_unix_(int filenum)
97 {
98  /*
99  * This makes sense only in the Unix world in which all negative
100  * integers different from -1 are non-canonical representations of
101  * the invalid socket descriptor.
102  */
103  if (!ight_socket_valid(filenum))
104  filenum = IGHT_SOCKET_INVALID;
105  return (filenum);
106 }
